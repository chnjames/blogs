var isTraditional = false;

function setTraditional(traditional) {
  isTraditional = traditional;
}

function convert(cc) {
  var str = '',
    ss = JTPYStr(),
    tt = FTPYStr();
  if (isTraditional) {
    ss = JTPYStr(),
      tt = FTPYStr();
  } else {
    tt = JTPYStr(),
      ss = FTPYStr();
  }
  for (var i = 0; i < cc.length; i++) {
    if (cc.charCodeAt(i) > 10000 && ss.indexOf(cc.charAt(i)) != -1) str += tt.charAt(ss.indexOf(cc.charAt(i)));
    else str += cc.charAt(i);
  }
  return str;
}

module.exports = {
  setTraditional: setTraditional,
  convert: convert
}

const convert = (cc, isTraditional) => {
  var str = '',
    ss = JTPYStr(),
    tt = FTPYStr();
  if (isTraditional) {
    ss = JTPYStr(),
      tt = FTPYStr();
  } else {
    tt = JTPYStr(),
      ss = FTPYStr();
  }
  for (var i = 0; i < cc.length; i++) {
    if (cc.charCodeAt(i) > 10000 && ss.indexOf(cc.charAt(i)) != -1) str += tt.charAt(ss.indexOf(cc.charAt(i)));
    else str += cc.charAt(i);
  }
  return str;
}

module.exports = {
  convert: convert
}

// 获取应用实例
const app = getApp()
const convertChs = require('../../utils/simp_trad_chs.js');

Page({
  data: {
    // 页面切换
    isToggled: false
  },
  // 切换简繁
  switchTab: function (e) {
    const index = e.currentTarget.dataset.index;
    const isTraditional = app.globalData.isTraditional;
    if ((index === 0 && !isTraditional) || (index === 1 && isTraditional)) {
      return;
    }
    app.globalData.isTraditional = !isTraditional;
    wx.setStorageSync('isTraditional', app.globalData.isTraditional);
    this.setData({
      isTraditional: app.globalData.isTraditional
    });
  },
  /**
   * 生命周期函数--监听页面加载
   */
  onLoad() {
    var that = this
    that.setData({
      isTraditional: app.globalData.isTraditional
    })
  },
  /**
   * 生命周期函数--监听页面显示
   */
  onShow: function () {
    const that = this
    if (that.data.isToggled) {
      that.onLoad()
    }
  },
  /**
   * 生命周期函数--监听页面隐藏
   */
  onHide: function () {
    this.setData({
      isToggled: true
    })
  }
})
